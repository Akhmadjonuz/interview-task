Introduction

This documentation aims to provide all the information you need to work with our API.

As you scroll, you'll see code examples for working with the API in different programming languages in the dark area to the right (or as part of the content on mobile). You can switch the language used with the tabs at the top right (or from the nav menu at the top left on mobile).

Base URL

http://task.beta

Authenticating requests

This API is not authenticated.

Authentication

Login user

Example request:
curl --request POST \
    "http://task.beta/api/login" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"phone\": 9,
    \"password\": \"aut\"
}"

const url = new URL(
    "http://task.beta/api/login"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "phone": 9,
    "password": "aut"
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());

Example response (200):


{
"result": [
 "id": 1,
 "phone": "793631747611",
 "role": 2,
 "token": "eyJ0eXA..."
]
}

Example response (400):


{
    "result": "Credential don't match or password error"
}

Product

Create product

requires authentication

Example request:
curl --request POST \
    "http://task.beta/api/product/create" \
    --header "Content-Type: multipart/form-data" \
    --header "Accept: application/json" \
    --form "name=qui" \
    --form "price=quod" \
    --form "category_id=4" \
    --form "brand_id=8" \
    --form "status=" \
    --form "photo=@C:\OSPanel\userdata\temp\php4C5E.tmp" 
const url = new URL(
    "http://task.beta/api/product/create"
);

const headers = {
    "Content-Type": "multipart/form-data",
    "Accept": "application/json",
};

const body = new FormData();
body.append('name', 'qui');
body.append('price', 'quod');
body.append('category_id', '4');
body.append('brand_id', '8');
body.append('status', '');
body.append('photo', document.querySelector('input[name="photo"]').files[0]);

fetch(url, {
    method: "POST",
    headers,
    body,
}).then(response => response.json());

Get product list

Example request:
curl --request GET \
    --get "http://task.beta/api/product/show?price=sunt&category_id=18&brand_id=3" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"price\": 10,
    \"category_id\": 8,
    \"brand_id\": 19
}"

const url = new URL(
    "http://task.beta/api/product/show"
);

const params = {
    "price": "sunt",
    "category_id": "18",
    "brand_id": "3",
};
Object.keys(params)
    .forEach(key => url.searchParams.append(key, params[key]));

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "price": 10,
    "category_id": 8,
    "brand_id": 19
};

fetch(url, {
    method: "GET",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
